3. 변수(값을 저장할 때 사용하는 식별자 입니다.)
   var 식별자;   또는   var 식별자=값;

   ex1) 
         var num1=100;
         var num2=200;
         var num3=num1+num2;
         alert('두수의 합은:'+num3);

   ex2)
        var radius, pi;
        radius=20;
        pi=3.1415926535;
        alert('원의 둘레:'+(2*pi*radius));

   ex3)복합연산자 (+=  -=  *=  /=  %=) 증감연산자(++  --)
       var result=100;
       result+=100;
       alert(result);
       result--;
       alert(result);

   ex4)
       window.onload = function () {

            // 문서에 출력합니다.
