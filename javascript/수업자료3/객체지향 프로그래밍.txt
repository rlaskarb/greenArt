    ex1)
     var product = {
            name: '7D 건조 망고',
            type: '당절임',

        alert(product.name+'\n'+product.type+'\n'+product.ingredient+'\n'+product.country);

     ex4) with 키워드를 사용한 간결한 표현

        var student = {
          이름: '연하진',
          국어: 92, 수학: 98,
          영어: 96, 과학: 98
        };
        with (student) {
            output += '이름: ' + 이름 + '\n';

            output += '수학: ' + 수학 + '\n';
            output += '영어: ' + 영어 + '\n';
            output += '과학: ' + 과학 + '\n';

        }
        alert(output);

   4)Date 객체

     ex1)
        var date = new Date();
        alert(date);

        *IE => UTC+0900시간으로 표시   그외 브라우저=>GMT+0900시간  (오차가 1초 미만)       

     ex2)
        var date = new Date();
        date.setDate(date.getDate() + 7);  // 현재 시간에서 7일을 더합니다.
        alert(date);

     ex3)
       var date = new Date();

        alert('년:'+date.getFullYear());
        alert('월:'+date.getMonth()+1);
        alert('일:'+date.getDate());
        alert('요일:'+date.getDay());   // 1~6=>월요일~토요일  0=>일요일
        alert('시:'+date.getHours());
        alert('분:'+date.getMinutes());
        alert('초:'+date.getSeconds());

   5)Math 객체

      ex1)
         alert(Math.abs(-800));   //절대값
         alert(Math.sqrt(100));   //제곱근
         alert(Math.floor(78.458));  //소수점이하 내림
         alert(Math.max(-800,400,30));  //최대값
         alert(Math.random()*10);  //0~1사이의 난수
         alert(Math.cos(50));  //코사인
         alert(Math.pow(10,2));  // 10*10 
      
         alert(Math.floor(Math.random()*10));  //0~10사이의 정수의 난수
