  $(document).ready(function() {
       var timeonoff;
       var imageCount=3;  
       var cnt=1;
       $('.btn1').addClass('clk1');
       $('.btn2').addClass('clk1');
       $('.btn3').addClass('clk1');
       $('.btn1').addClass('clk2');
       $('.visibleBox .list1').fadeIn('slow');	

       function timer_gallery(){
           cnt++;
           for(var i=1;i<=imageCount;i++){
              $('.visibleBox .list'+i).hide();
           }
          $('.visibleBox .list'+cnt).fadeIn('slow');
		
          for(var i=1;i<=imageCount;i++){
               $('.btn'+i).removeClass('clk2');
          }
           $('.btn'+cnt).addClass('clk2'); 
            if(cnt==imageCount)cnt=0;
      }
     timeonoff= setInterval(timer_gallery, 4000);
	
	 
    $('.selectBox span').each(function (index) {
      $(this).click(function(){
          clearInterval(timeonoff);
          for(var i=1;i<=imageCount;i++){
               $('.visibleBox .list'+i).hide();
          }
          $('.visibleBox .list'+(index+1)).fadeIn('slow'); 
      
         for(var i=1;i<=imageCount;i++){
            $('.btn'+i).removeClass('clk2');
          }
         $('.btn'+(index+1)).addClass('clk2');  
          cnt=index+1;
          if(cnt==imageCount)cnt=0;
          timeonoff= setInterval(timer_gallery, 4000);
          });
       });

	var myArticle = $(this).parents('.article'); 
	
	if(myArticle.hasClass('hide')){   
	    article.find('.a').slideUp(100);
                  article.removeClass('show').addClass('hide'); 
	    myArticle.removeClass('hide').addClass('show');  
	    myArticle.find('.a').slideDown(100);  
	 } else {  
	   myArticle.removeClass('show').addClass('hide');  
	   myArticle.find('.a').slideUp(100);  
	}  
  });       
});

    $(document).ready(function() {
        ticker_height = $('#news_scroller').height();  
        news_height=$('#news_scroller p').height();
        no_oflines = 0;
        rotate = true;

       $('#news_scroller').hover(
       function(){   
          rotate = false;
       }, 
       function(){
          rotate = true;
       });
      scroll();
    });

    function scroll() {
      no_oflines += rotate ? -2 : 0;    //마우스가 아웃되어 있으면..2씩 감소, 올라가면 멈춤
      $('#news_scroller p').css( 'top', ''+no_oflines+"px" );
      if( no_oflines<-1*news_height ){  //위로 올라가 완전히 사라지면..
          no_oflines = ticker_height;   //아래로 내려간다.
      }
      setTimeout( scroll, 50 );  //재귀함수
    }

$(document).ready(function() {
  var timeonoff;
  var imageCount=5;  
  var cnt=1;
  var direct=1;
  var position=0;
  $('.btn1').css('background','url(images/onbtn1.png)'); 

 function moveg(){
      cnt+=direct;
	
		if(cnt==1){
		   position=0;
		}else if(cnt==2){
	               position=-1000;
		}else if(cnt==3){
		   position=-2000;
		}else if(cnt==4){
		   position=-3000;
		}else if(cnt==5){
		   position=-4000;
		}
                                                
                         for(var i=1;i<=imageCount;i++){
                            $('.btn'+i).css('background','url(images/'+i+'.png)'); 
                         }
                         $('.btn'+cnt).css('background','url(images/onbtn'+cnt+'.png)'); 
                           
		$('.gallery').animate({left:position}, 'slow');
		if(cnt==imageCount)direct=-1;
                         if(cnt==1)direct=1;
 }

  timeonoff= setInterval( moveg , 4000);

  $('.mbutton').click(function(event){

	var $target=$(event.target);
            clearInterval(timeonoff); 

            for(var i=1;i<=imageCount;i++){
                $('.btn'+i).css('background','url(images/'+i+'.png)'); 
            }

	if($target.is('.btn1')){
    	   $('.gallery').animate({left:0}, 'slow');
                cnt=1;
                direct=1;
	}else if($target.is('.btn2')){
    	   $('.gallery').animate({left:-1000}, 'slow');
                cnt=2;
	}else if($target.is('.btn3')){
    	   $('.gallery').animate({left:-2000}, 'slow');
                cnt=3;
	}else if($target.is('.btn4')){
    	   $('.gallery').animate({left:-3000}, 'slow');
                cnt=4;
	}else if($target.is('.btn5')){
    	   $('.gallery').animate({left:-4000}, 'slow');
                cnt=5;
                direct=-1;
	}
              $('.btn'+cnt).css('background','url(images/onbtn'+cnt+'.png)'); 
           timeonoff= setInterval( moveg , 4000);
  });
});

$(document).ready(function(){
  var cnt=3;
  $('.tabs .contlist:eq(0)').show(); 
  $('.tabs .tab1').css('background','url(images/tab1on.jpg)');
  
  $('.tabs .tab').each(function (index) {
    $(this).click(function(){    
	  $(".tabs .contlist").hide(); 
	  $(".tabs .contlist:eq("+index+")").show();
	  for(var i=1;i<=cnt;i++){
           $('.tab'+i).css('background','url(images/tab'+i+'.jpg)');
      }
      $(this).css('background','url(images/tab'+(index+1)+'on.jpg)'); 
   });
  });
});
